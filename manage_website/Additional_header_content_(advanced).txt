var arrayOfPanels = ["A_C", "D_G", "H_L", "M_R", "S_V", "W_Z"];

/* function to show and hide directory sections */
function showDirectoryPanel(panelToShow){
	for(i=0; i<arrayOfPanels.length; i++){
		document.getElementById(arrayOfPanels[i]).style.display = "none";
document.getElementById(arrayOfPanels[i]).parentElement.parentElement.parentElement.style.display = "none";
document.getElementById("anchor"+arrayOfPanels[i]).parentElement.className = "";
	}	
document.getElementById(panelToShow).style.display = "block";
document.getElementById(panelToShow).parentElement.parentElement.parentElement.style.display = "block";
document.getElementById("anchor"+panelToShow).parentElement.className = "active";
}

/* Function to add event handlers onto the directory links*/
function addEventListenersToDirectoryLinks(){
	document.getElementById("anchorA_C").addEventListener("click", function(){ showDirectoryPanel("A_C") });
	document.getElementById("anchorD_G").addEventListener("click", function(){ showDirectoryPanel("D_G") });
	document.getElementById("anchorH_L").addEventListener("click", function(){ showDirectoryPanel("H_L") });
	document.getElementById("anchorM_R").addEventListener("click", function(){ showDirectoryPanel("M_R") });
